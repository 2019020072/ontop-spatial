NAME
        ontop query execute - Executes a query against a database; prints the
        query results on STDOUT.

SYNOPSIS
        ontop query execute [ {-b | --bind} <variables to bind>... ]
                [ {-f | --format} <resultFormat> ] [ {-p | --passwd} <password> ]
                [ {-P | --ask-password} ] [ {-r | --reasoning} ]
                [ {-u | --username} <username> ] [ {-v | --verbose} ] [--] <database>
                <query>...
